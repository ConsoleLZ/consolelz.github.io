// 判读照片链接的函数
function checkImg(imgUrl) {
    return new Promise((resolve, reject) => {
        var ImgObj = new Image()
        ImgObj.src = imgUrl
        ImgObj.onload = function(res) {
            resolve(res)
        }
        ImgObj.onerror = function(err) {
            reject(err)
        }
    })
}
